利用递归实现快速排序

近日在课上学习了冒泡排序法，联想到最近学习的函数的递归，
于是利用递归写了一个排序程序，代码如下：
#include<stdio.h>
#define N 10
void sort (int a[], int low, int high);
int split (int a[],int low,int high);

int main()
{
int a[N],i;
for(i=0;i<N;i++)
scanf(“%d”,&a[i]);

sort(a,0,N-1);

for(i=0;i<N;i++)
    printf(" %d",a[i]);
printf("\n");

return 0;

}

void sort (int a[],int low,int high)
{
int middle;

if (low>high)
return;

middle=split(a,low,high);


sort(a,low,middle-1);
sort(a,middle+1,high);

}

int split(int a[],int low,int high)
{
int part_element=a[low];

for(;;){
    
    while(low<high&&part_element<=a[high])
        high--;
    
    if(low>=high)
    break;
    
    a[low]=a[high];
    low++;
    while(low<high&&a[low]<=part_element)
        low++;
        
    if(low>=high)
    break;
    
    a[high]=a[low];
    high--;
}

a[high]=part_element;
return high;

}
代码定义了排序函数sort和分割数列函数split，先通过split得出
一个元素，利用这个元素将函数分割成两段，然后利用递归在两
段进行排序。
